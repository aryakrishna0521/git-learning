git config --list [list user configuration]

git config --global user.name "username" [set username]

git config --global user.email "useremail" [set user email]

git init         [initialize git in cwf]

git add -A        [adding all changes to staging area]

git commit -m "commit message"

git remote add remote_name path    [for adding remote repository] 
eg:git remote add origin https://github.com/aryakrishna0521/html.git

git push remote_name branch
git branch   [to list branch]
eg:git push origin master
